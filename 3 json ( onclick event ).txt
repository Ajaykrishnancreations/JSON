<!DOCTYPE html>
<html lang="en">
<head>
    
    <title>Json</title>
    
    <script>
        var v1=new XMLHttpRequest();
        v1.onreadystatechange=function(){
            if(this.readyState==4 && this.status==200){
                
                var v2=JSON.parse(v1.responseText);
                
                //console.log(v2.a);
                var v3=v2.a;
                var t=" ";
                    for(var i=0;i<v3.length;i++)
                        {
                            t+="<li>"+v3[i].name+" "+v3[i].age+"</li>"
                        }

                        document.getElementById("b").addEventListener("click",n)
                        function n(){
                            document.getElementById("o").innerHTML=t;
                        }
            }
        };
        v1.open("GET","data.json",true);
        v1.send();
    </script>

</head>
<body>
    <ol id="o"></ol>
    <button id="b">click here to display the json list</button>
</body>
</html>
